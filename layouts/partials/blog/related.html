Fallback: top 3 posts in /blog/ with the most overlap on `topics:`, excluding the current post. */ -}} {{- $current := . -}} {{- $lang := .Params.language | default "en" -}} {{- $related := slice -}} {{- with .Params.related_posts -}} {{- range . -}} {{- with $.Site.GetPage . -}}{{- $related = $related | append . -}}{{- end -}} {{- end -}} {{- end -}} {{- if not $related -}} {{- $myTopics := $current.Params.topics | default slice -}} {{- if $myTopics -}} {{- $candidates := where $.Site.RegularPages "Section" "blog" -}} {{- $candidates = where $candidates "Permalink" "ne" $current.Permalink -}} {{- $scored := slice -}} {{- range $candidates -}} {{- $other := . -}} {{- $otherTopics := .Params.topics | default slice -}} {{- $score := 0 -}} {{- range $myTopics -}} {{- if in $otherTopics . -}}{{- $score = add $score 1 -}}{{- end -}} {{- end -}} {{- if gt $score 0 -}} {{- $scored = $scored | append (dict "page" $other "score" $score) -}} {{- end -}} {{- end -}} {{- $scored = sort $scored "score" "desc" -}} {{- range first 3 $scored -}} {{- $related = $related | append .page -}} {{- end -}} {{- end -}} {{- end -}} {{- /* Final fallback: 3 newest posts */ -}} {{- if not $related -}} {{- $latest := where $.Site.RegularPages "Section" "blog" -}} {{- $latest = where $latest "Permalink" "ne" $current.Permalink -}} {{- $latest = $latest.ByDate.Reverse -}} {{- range first 3 $latest -}}{{- $related = $related | append . -}}{{- end -}} {{- end -}} {{- if $related -}} {{- $heading := "Read also" -}} {{- if eq $lang "de" -}}{{- $heading = "Lesen Sie auch" -}}{{- end -}}
